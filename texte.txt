from services.authentification import option_user
from config.constants import MENU_PRINCIPALE,MENU_connect,CHOIX_PRINCIPAL, OPTION_PRINCIPALE
from config.constants import GESTION_DES_ETUDIANTS,GESTION_DES_PROFFESSEURS,GESTION_DES_UTILISATEURS,GESTION_DES_MATIERES,GESTION_DES_NOTES,GESTION_DES_ABSENCES
from models.users import userModel


def initialiser_bdd():
    liaison = userModel()
    liaison.users()
    liaison.close()


def main():

    initialiser_bdd()

    compte_connecte = option_user()

    if compte_connecte:

        nom_du_gars = compte_connecte[1]

        while True:

            choix = input(f"{nom_du_gars}, choisissez une action : ")

            if choix == '1':
               print(GESTION_DES_ETUDIANTS)

            elif choix == '2':
                print(GESTION_DES_PROFFESSEURS)

            elif choix == '3':
                print(GESTION_DES_UTILISATEURS)
                
            elif choix == "4":
                print(GESTION_DES_MATIERES)

            elif choix == "5":
                print(GESTION_DES_NOTES)

            elif choix == "6":
                print(GESTION_DES_ABSENCES)

            elif choix == "0":
                print(MENU_connect)
                break

            else:
                print("Option invalide, veuillez réessayer.")


if __name__ == "__main__":
    main()
